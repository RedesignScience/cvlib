@dataclass(frozen=True)
class GlobalParameter(Serializable):
    """
    A global parameter to be used in OpenMM simulations.

    Parameters
    ----------
    name
        The name of the parameter.
    default
        The default value of the parameter.
    unit
        The unit of measurement of the global parameter. If it does not have a unit,
        use ``unit.dimensionless``.
    """

    name: str
    default: Real
    unit: mmunit.Unit

    def __post_init__(self):
        if not isinstance(self.name, str):
            raise TypeError("The 'name' argument must be a string.")
        if not isinstance(self.default, Real):
            raise TypeError("The 'default' argument must be a real number.")
        if not mmunit.is_unit(self.unit):
            raise ValueError("The 'unit' argument must be a valid, MD-compatible unit.")

    def setValue(self, context: openmm.Context, value: ScalarQuantity) -> None:
        """
        Set the value of the global parameter in the given OpenMM context object.
        Unit conversion is performed if necessary.

        Parameters
        ----------
        context
            The OpenMM context object.
        value
            The new value of the global parameter.
        """
        if mmunit.is_quantity(value):
            value = value.value_in_unit(self.unit)
        context.setParameter(self.name, value)

    def getValue(self, context: openmm.Context) -> Quantity:
        """
        Get the current value of the global parameter.

        Parameters
        ----------
        context
            The OpenMM context object.

        Returns
        -------
        value
            The current value of the global parameter.
        """
        return context.getParameter(self.name) * self.unit
