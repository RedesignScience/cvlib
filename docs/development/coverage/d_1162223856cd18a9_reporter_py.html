21class Reporter(mmapp.StateDataReporter): 

22 """ 

23 Reports values and/or effective masses of collective variables during an OpenMM 

24 `Simulation`_. 

25 

26 Create a :class:`Reporter` add it to the `Simulation`_'s list 

27 of reporters (see example below). The reporter writes data to a file or file-like 

28 object at regular intervals. The set of data to write is configurable using boolean 

29 flags passed to the constructor. The data is written in comma-separated-value (CSV) 

30 format by default, but the user can specify a different separator. 

31 

32 .. _Simulation: http://docs.openmm.org/latest/api-python/generated/ 

33 openmm.app.simulation.Simulation.html 

34 

35 Parameters 

36 ---------- 

37 file 

38 The file to write to. This can be a file name or a file object. 

39 reportInterval 

40 The interval (in time steps) at which to report data. 

41 variables 

42 The collective variable whose values and/or effective masses will be reported. 

43 These must be the same objects added to the simulation's :OpenMM:`System`. 

44 metaCVs 

45 The meta-collective variables whose inner variables will be reported. These must 

46 be the same objects added to the simulation's :OpenMM:`System`. 

47 step 

48 Whether to report the current step index. 

49 time 

50 Whether to report the current simulation time. 

51 values 

52 Whether to report the current values of the collective variables. 

53 masses 

54 Whether to report the current effective masses of the collective variables. 

55 parameters 

56 The names of metaCV parameters to report. 

57 separator 

58 The separator to use between columns in the file. 

59 append 

60 If `True`, omit the header line and append the report to an existing file. 

61 

62 Raises 

63 ------ 

64 ValueError 

65 If `variables` and `metaCVs` are both empty. 

66 ValueError 

67 If `values` and `masses` are both `False`. 

68 ValueError 

69 If any value `contextParameters` is not a unit of measurement. 

70 

71 Examples 

72 -------- 

73 >>> import cvpack 

74 >>> import openmm 

75 >>> from openmm import app, unit 

76 >>> from sys import stdout 

77 >>> from openmmtools import testsystems 

78 >>> model = testsystems.AlanineDipeptideVacuum() 

79 >>> phi = cvpack.Torsion(6, 8, 14, 16, name="phi") 

80 >>> psi = cvpack.Torsion(8, 14, 16, 18, name="psi") 

81 >>> umbrella = cvpack.MetaCollectiveVariable( 

82 ... "0.5*kappa*(min(dphi,2*pi-dphi)^2+min(dpsi,2*pi-dpsi)^2)" 

83 ... "; dphi=abs(phi-5*pi/6); dpsi=abs(psi+5*pi/6)", 

84 ... [phi, psi], 

85 ... unit.kilojoules_per_mole, 

86 ... name="umbrella", 

87 ... pi=3.141592653589793, 

88 ... kappa=100 * unit.kilojoules_per_mole/unit.radian**2, 

89 ... ) 

90 >>> reporter = cvpack.Reporter( 

91 ... stdout, 

92 ... 100, 

93 ... variables=[umbrella], 

94 ... metaCVs=[umbrella], 

95 ... step=True, 

96 ... values=True, 

97 ... masses=True, 

98 ... parameters=["kappa"], 

99 ... ) 

100 >>> integrator = openmm.LangevinIntegrator( 

101 ... 300 * unit.kelvin, 

102 ... 1 / unit.picosecond, 

103 ... 2 * unit.femtosecond, 

104 ... ) 

105 >>> integrator.setRandomNumberSeed(1234) 

106 >>> umbrella.addToSystem(model.system) 

107 >>> simulation = app.Simulation(model.topology, model.system, integrator) 

108 >>> simulation.context.setPositions(model.positions) 

109 >>> simulation.context.setVelocitiesToTemperature(300 * unit.kelvin, 5678) 

110 >>> simulation.reporters.append(reporter) 

111 >>> simulation.step(1000) # doctest: +SKIP 

112 #"Step","umbrella (kJ/mol)",...(nm**2 Da/(rad**2))","kappa (kJ/(mol rad**2))"

125 def __init__( # pylint: disable=dangerous-default-value 

126 self, 

127 file: t.Union[str, io.TextIOBase], 

128 reportInterval: int, 

129 variables: t.Sequence[CollectiveVariable] = (), 

130 metaCVs: t.Sequence[MetaCollectiveVariable] = (), 

131 step: bool = False, 

132 time: bool = False, 

133 values: bool = False, 

134 masses: bool = False, 

135 parameters: t.Iterable[str] = (), 

136 separator: str = ",", 

137 append: bool = False, 

138 ) -> None: 

139 super().__init__( 

140 file, 

141 reportInterval, 

142 step=step, 

143 time=time, 

144 separator=separator, 

145 append=append, 

146 ) 

147 self._variables = variables 

148 self._meta_cvs = metaCVs 

149 self._values = values 

150 self._masses = masses 

151 self._parameter_unit_pairs = { 

152 (name, unit) 

153 for metacv in self._meta_cvs 

154 for name, unit in metacv.getParameterUnits().items() 

155 if name in parameters 

156 } 

157 self._parameter_units = dict(self._parameter_unit_pairs) 

158 self._validate() 

159 

160 def _validate(self) -> None: 

161 if not (self._variables or self._meta_cvs): 

162 raise ValueError("Arguments 'variables' and 'metaCVs' cannot be both empty") 

163 if not (self._values or self._masses): 

164 raise ValueError("Arguments 'values' and 'masses' cannot be both False") 

165 for cv in self._variables: 

166 if not isinstance(cv, CollectiveVariable): 

167 raise TypeError( 

168 "All items in 'variables' must be instances of CollectiveVariable" 

169 ) 

170 for metacv in self._meta_cvs: 

171 if not isinstance(metacv, MetaCollectiveVariable): 

172 raise TypeError( 

173 "All items in 'metaCVs' must be instances of MetaCollectiveVariable" 

174 ) 

175 if len(self._parameter_unit_pairs) < len(self._parameter_units): 

176 raise ValueError("All items in 'parameters' must be present in 'metaCVs'") 

177 if len(self._parameter_unit_pairs) > len(self._parameter_units): 

178 raise ValueError("Parameters must have consistent units across 'metaCVs'") 

179 

180 def _constructReportValues( # pylint: disable=too-many-branches 

181 self, simulation: mmapp.Simulation, state: openmm.State 

182 ) -> t.List[float]: 

183 values = [] 

184 if self._step: 

185 values.append(simulation.currentStep) 

186 if self._time: 

187 values.append(state.getTime().value_in_unit(mmunit.picosecond)) 

188 context = simulation.context 

189 for cv in self._variables: 

190 if self._values: 

191 values.append(cv.getValue(context) / cv.getUnit()) 

192 if self._masses: 

193 values.append(cv.getEffectiveMass(context) / cv.getMassUnit()) 

194 for cv in self._meta_cvs: 

195 cv_values = cv.getInnerValues(context) if self._values else {} 

196 cv_masses = cv.getInnerEffectiveMasses(context) if self._masses else {} 

197 for name in cv_values if self._values else cv_masses: 

198 if name in cv_values: 

199 values.append(cv_values[name] / cv_values[name].unit) 

200 if name in cv_masses: 

201 values.append(cv_masses[name] / cv_masses[name].unit) 

202 for parameter in self._parameter_units: 

203 values.append(context.getParameter(parameter)) 

204 return values 

205 

206 def _constructHeaders(self) -> t.List[str]: 

207 headers = [] 

208 if self._step: 

209 headers.append("Step") 

210 if self._time: 

211 headers.append("Time (ps)") 

212 

213 def add_headers(cv: CollectiveVariable) -> None: 

214 if self._values: 

215 headers.append(f"{cv.getName()} ({cv.getUnit().get_symbol()})") 

216 if self._masses: 

217 headers.append(f"{cv.getName()} mass ({cv.getMassUnit().get_symbol()})") 

218 

219 for cv in self._variables: 

220 add_headers(cv) 

221 for cv in self._meta_cvs: 

222 for inner_cv in cv.getInnerVariables(): 

223 add_headers(inner_cv) 

224 for parameter, unit in self._parameter_units.items(): 

225 headers.append(f"{parameter} ({unit.get_symbol()})") 

226 return headers
