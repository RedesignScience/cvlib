14import openmm

17from .path import Metric, deviation, progress 

18from .units.units import MatrixQuantity, ScalarQuantity, value_in_md_units

22class PathInCVSpace(openmm.CustomCVForce, CollectiveVariable): 

23 r""" 

24 A metric of the system's progress (:math:`s`) or deviation (:math:`z`) with 

25 respect to a path defined by a sequence of :math:`n` milestones positioned in a 

26 collective variable space :cite:`Branduardi_2007`: 

27 

28 .. math:: 

29 

30 s({\bf r}) = \frac{ 

31 \dfrac{\sum_{i=1}^n i w_i({\bf r})}{\sum_{i=1}^n w_i({\bf r})} - 1 

32 }{n-1} 

33 \quad \text{or} \quad 

34 z({\bf r}) = - 2 \sigma ^2 \ln \sum_{i=1}^n w_i({\bf r}) 

35 

36 with :math:`w_i({\bf r})` being a Gaussian kernel centered at the :math:`i`-th 

37 milestone, i.e., 

38 

39 .. math:: 

40 

41 w_i({\bf r}) = \exp\left(\ 

42 -\frac{\|{\bf c}({\bf r}) - \hat{\bf c}_i\|^2}{2 \sigma^2} 

43 \right) 

44 

45 where :math:`{\bf c}({\bf r})` is a vector of collective variables, 

46 :math:`\hat{\bf c}_i` is the location of the :math:`i`-th milestone, and 

47 :math:`\sigma` sets the width of the kernels. The squared norm of a vector 

48 :math:`{\bf x}` in the collective variable space is defined as 

49 

50 .. math:: 

51 

52 \|{\bf x}\|^2 = {\bf x}^T {\bf D}^{-2} {\bf x} 

53 

54 where :math:`{\bf D}` is a diagonal matrix whose each diagonal element is the 

55 characteristic scale of the corresponding collective variable, which makes 

56 :math:`\|{\bf x}\|^2` dimensionless. Appropriate boundary conditions are used for 

57 periodic CVs. 

58 

59 .. note:: 

60 

61 The kernel width :math:`\sigma` is related to the parameter :math:`\lambda` of 

62 Ref. :cite:`Branduardi_2007` by :math:`\sigma = \frac{1}{\sqrt{2\lambda}}`. 

63 

64 Parameters 

65 ---------- 

66 metric 

67 The path-related metric to compute. Use ``cvpack.path.progress`` for 

68 computing :math:`s({\bf r})` or ``cvpack.path.deviation`` for computing 

69 :math:`z({\bf r})`. 

70 variables 

71 The collective variables that define the space. 

72 milestones 

73 The milestones in the collective variable space. The number of rows must be 

74 equal to the number of milestones and the number of columns must be equal to 

75 the number of collective variables. 

76 sigma 

77 The width of the Gaussian kernels. 

78 scales 

79 The characteristic scales for the collective variables. If not provided, the 

80 scales are assumed to be 1 (in standard MD units) for each collective variable. 

81 name 

82 The name of the collective variable. If not provided, it is set to 

83 "path_progress_in_cv_space" or "path_deviation_in_cv_space" depending on the 

84 metric. 

85 

86 Raises 

87 ------ 

88 ValueError 

89 If the number of rows in the milestones matrix is less than 2 

90 ValueError 

91 If the number of columns in the milestones matrix is different from the number 

92 of collective variables 

93 ValueError 

94 If the metric is not `cvpack.path.progress` or `cvpack.path.deviation` 

95 

96 Examples 

97 -------- 

98 >>> import cvpack 

99 >>> from openmmtools import testsystems 

100 >>> import numpy as np 

101 >>> model = testsystems.AlanineDipeptideVacuum() 

102 >>> phi_atoms = ["ACE-C", "ALA-N", "ALA-CA", "ALA-C"] 

103 >>> psi_atoms = ["ALA-N", "ALA-CA", "ALA-C", "NME-N"] 

104 >>> atoms = [f"{a.residue.name}-{a.name}" for a in model.topology.atoms()] 

105 >>> milestones = np.array( 

106 ... [[1.3, -0.2], [1.2, 3.1], [-2.7, 2.9], [-1.3, 2.7], [-1.3, -0.4]] 

107 ... ) 

108 >>> phi = cvpack.Torsion(*[atoms.index(atom) for atom in phi_atoms]) 

109 >>> psi = cvpack.Torsion(*[atoms.index(atom) for atom in psi_atoms]) 

110 >>> path_vars = [] 

111 >>> for metric in (cvpack.path.progress, cvpack.path.deviation): 

112 ... var = cvpack.PathInCVSpace(metric, [phi, psi], milestones, np.pi / 6) 

113 ... var.addToSystem(model.system) 

114 ... path_vars.append(var) 

115 >>> context = openmm.Context(model.system, openmm.VerletIntegrator(1.0)) 

116 >>> context.setPositions(model.positions) 

117 >>> path_vars[0].getValue(context) 

118 0.50... dimensionless 

119 >>> path_vars[1].getValue(context) 

120 0.25... dimensionless 

121 """ 

122 

123 def __init__( # pylint: disable=too-many-branches 

124 self, 

125 metric: Metric, 

126 variables: t.Iterable[CollectiveVariable], 

127 milestones: MatrixQuantity, 

128 sigma: float, 

129 scales: t.Optional[t.Iterable[ScalarQuantity]] = None, 

130 name: t.Optional[str] = None, 

131 ) -> None: 

132 if metric not in (progress, deviation): 

133 raise ValueError( 

134 "Invalid metric. Use 'cvpack.path.progress' or 'cvpack.path.deviation'." 

135 ) 

136 if name is None: 

137 name = f"path_{metric.name}_in_cv_space" 

138 variables = list(variables) 

139 cv_scales = [1.0] * len(variables) if scales is None else list(scales) 

140 milestones, n, numvars = convert_to_matrix(milestones) 

141 if numvars != len(variables): 

142 raise ValueError("Wrong number of columns in the milestones matrix.") 

143 if n < 2: 

144 raise ValueError("At least two rows are required in the milestones matrix.") 

145 definitions = OrderedDict({"lambda": 1 / (2 * sigma**2)}) 

146 periods = {} 

147 for i, variable in enumerate(variables): 

148 values = variable.getPeriodicBounds() 

149 if values is not None: 

150 periods[i] = value_in_md_units(values[1] - values[0]) 

151 for i, values in enumerate(milestones): 

152 deltas = [f"({value}-cv{j})" for j, value in enumerate(values)] 

153 for j, period in periods.items(): 

154 deltas[j] = f"min(abs{deltas[j]},{period}-abs{deltas[j]})" 

155 definitions[f"x{i}"] = "+".join( 

156 f"{delta}^2" if scale == 1.0 else f"({delta}/{scale})^2" 

157 for delta, scale in zip(deltas, cv_scales) 

158 ) 

159 definitions["xmin0"] = "min(x0,x1)" 

160 for i in range(n - 2): 

161 definitions[f"xmin{i+1}"] = f"min(xmin{i},x{i+2})" 

162 for i in range(n): 

163 definitions[f"w{i}"] = f"exp(lambda*(xmin{n - 2}-x{i}))" 

164 definitions["wsum"] = "+".join(f"w{i}" for i in range(n)) 

165 expressions = [f"{key}={value}" for key, value in definitions.items()] 

166 if metric == progress: 

167 numerator = "+".join(f"{i}*w{i}" for i in range(1, n)) 

168 expressions.append(f"({numerator})/({n - 1}*wsum)") 

169 else: 

170 expressions.append(f"xmin{n - 2}-log(wsum)/lambda") 

171 super().__init__("; ".join(reversed(expressions))) 

172 for i, variable in enumerate(variables): 

173 self.addCollectiveVariable(f"cv{i}", deepcopy(variable)) 

174 self._registerCV( 

175 name, 

176 None, 

177 metric, 

178 variables, 

179 milestones.tolist(), 

180 sigma, 

181 scales, 

182 ) 

183 

184 

185PathInCVSpace.registerTag("!cvpack.PathInCVSpace")
