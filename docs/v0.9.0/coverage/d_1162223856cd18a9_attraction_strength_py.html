22class AttractionStrength(openmm.CustomNonbondedForce, BaseCollectiveVariable): 

23 r""" 

24 The strength of the attraction between two atom groups: 

25 

26 .. math:: 

27 

28 S_{\rm attr}({\bf r}) = S_{1,2}({\bf r}) = &-\frac{1}{E_{\rm ref}} \Bigg[ 

29 \sum_{i \in {\bf g}_1} 

30 \sum_{\substack{j \in {\bf g}_2 \\ j \neq i}} 

31 \epsilon_{ij} u_{\rm disp}\left( 

32 \frac{\|{\bf r}_i - {\bf r}_j\|}{\sigma_{ij}} 

33 \right) \\ 

34 &+\sum_{i \in {\bf g}_1} 

35 \sum_{\substack{j \in {\bf g}_2 \\ q_jq_i < 0}} 

36 \frac{q_i q_j}{4 \pi \varepsilon_0 r_{\rm c}} u_{\rm elec}\left( 

37 \frac{\|{\bf r}_i - {\bf r}_j\|}{r_{\rm c}} 

38 \right) \Bigg] 

39 

40 where :math:`{\bf g}_1` and :math:`{\bf g}_2` are the two atom groups, 

41 :math:`r_{\rm c}` is the cutoff distance, :math:`\varepsilon_0` is the 

42 permittivity of empty space, :math:`q_i` is the charge of atom :math:`i`, and 

43 :math:`E_{\rm ref}` is a reference value (in energy units per mole). 

44 The Lennard-Jones parameters are given by the Lorentz-Berthelot mixing rule, i.e. 

45 :math:`\epsilon_{ij} = \sqrt{\epsilon_i \epsilon_j}`, and 

46 :math:`\sigma_{ij} = (\sigma_i + \sigma_j)/2`. 

47 

48 Optionally, one can provide a third atom group :math:`{\bf g}_3` to contrast 

49 the attraction strength between :math:`{\bf g}_1` and :math:`{\bf g}_2` with 

50 that between :math:`{\bf g}_1` and :math:`{\bf g}_3`. In this case, the 

51 collective variable becomes 

52 

53 .. math:: 

54 

55 S_{\rm attr}({\bf r}) = S_{1,2}({\bf r}) - S_{1,3}({\bf r}) 

56 

57 .. note:: 

58 

59 Groups :math:`{\bf g}_1` and :math:`{\bf g}_2` can overlap or even be the 

60 same, in which case the collective variable will measure the strength of 

61 the self-attraction of :math:`{\bf g}_1`. On the other hand, the contrast 

62 group :math:`{\bf g}_3` cannot overlap with neither :math:`{\bf g}_1` nor 

63 :math:`{\bf g}_2`. 

64 

65 The function :math:`u_{\rm disp}(x)` is a Lennard-Jones-like reduced potential with 

66 a highly softened repulsion part, defined as 

67 

68 .. math:: 

69 

70 u_{\rm disp}(x) = 4\left(\frac{1}{y^2} - \frac{1}{y}\right), 

71 \quad \text{where} \quad 

72 y = |x^6 - 2| + 2 

73 

74 The function :math:`u_{\rm elec}(x)` provides a Coulomb-like decay with 

75 reaction-field screening, defined as 

76 

77 .. math:: 

78 

79 u_{\rm elec}(x) = \frac{1}{x} + \frac{x^2 - 3}{2} 

80 

81 The screening considers a perfect conductor as the surrounding medium 

82 :cite:`Correa_2022`. 

83 

84 .. note:: 

85 

86 Only attractive electrostatic interactions are considered (:math:`q_i q_i < 0`), 

87 which gives :math:`S_{\rm attr}({\bf r})` a lower bound of zero. The upper 

88 bound will depends on the system details, the chosen groups of atoms, and the 

89 adopted reference value. 

90 

91 The Lennard-Jones parameters, atomic charges, cutoff distance, boundary conditions, 

92 as well as whether to use a switching function and its corresponding switching 

93 distance, are taken from :openmm:`NonbondedForce` object. 

94 

95 .. note:: 

96 Any non-exclusion exceptions involving an atom in :math:`{\bf g}_1` and an atom 

97 in either :math:`{\bf g}_2` or :math:`{\bf g}_3` are turned into exclusions in 

98 this collective variable. 

99 

100 Parameters 

101 ---------- 

102 group1 

103 The first atom group. 

104 group2 

105 The second atom group. 

106 nonbondedForce 

107 The :class:`openmm.NonbondedForce` object from which to collect the necessary 

108 parameters. 

109 reference 

110 A reference value (in energy units per mole) to which the collective variable 

111 should be normalized. One can also provide an :OpenMM:`Context` object from 

112 which to obtain a reference attraction strength. 

113 contrastGroup 

114 An optional third atom group to contrast the attraction strength between 

115 :math:`{\bf g}_1` and :math:`{\bf g}_2` with that between :math:`{\bf g}_1` 

116 and :math:`{\bf g}_3`. 

117 

118 Examples 

119 -------- 

120 >>> import cvpack 

121 >>> from openmm import unit 

122 >>> from openmmtools import testsystems 

123 >>> model = testsystems.HostGuestExplicit() 

124 >>> guest = [a.index for a in model.topology.atoms() if a.residue.name == "B2"] 

125 >>> host = [a.index for a in model.topology.atoms() if a.residue.name == "CUC"] 

126 >>> forces = {f.getName(): f for f in model.system.getForces()} 

127 >>> cv1 = cvpack.AttractionStrength(guest, host, forces["NonbondedForce"]) 

128 >>> _ = cv1.setUnusedForceGroup(0, model.system) 

129 >>> _ = model.system.addForce(cv1) 

130 >>> platform = openmm.Platform.getPlatformByName("Reference") 

131 >>> integrator = openmm.VerletIntegrator(1.0 * mmunit.femtoseconds) 

132 >>> context = openmm.Context(model.system, integrator, platform) 

133 >>> context.setPositions(model.positions) 

134 >>> print(cv1.getValue(context, 4)) 

135 4912.5 dimensionless 

136 

137 >>> water = [a.index for a in model.topology.atoms() if a.residue.name == "HOH"] 

138 >>> cv2 = cvpack.AttractionStrength(guest, water, forces["NonbondedForce"]) 

139 >>> _ = cv2.setUnusedForceGroup(0, model.system) 

140 >>> _ = model.system.addForce(cv2) 

141 >>> context.reinitialize(preserveState=True) 

142 >>> print(cv2.getValue(context, 4)) 

143 2063.3 dimensionless 

144 

145 >>> cv3 = cvpack.AttractionStrength( 

146 ... guest, host, forces["NonbondedForce"], contrastGroup=water, 

147 ... ) 

148 >>> _ = cv3.setUnusedForceGroup(0, model.system) 

149 >>> _ = model.system.addForce(cv3) 

150 >>> context.reinitialize(preserveState=True) 

151 >>> print(cv3.getValue(context, 4)) 

152 2849.2 dimensionless 

153 >>> print(cv1.getValue(context, 4) - cv2.getValue(context, 4)) 

154 2849.2 dimensionless 

155 """

159 @mmunit.convert_quantities 

160 def __init__( # pylint: disable=too-many-arguments 

161 self, 

162 group1: t.Iterable[int], 

163 group2: t.Iterable[int], 

164 nonbondedForce: openmm.NonbondedForce, 

165 reference: t.Union[mmunit.ScalarQuantity, openmm.Context] = mmunit.Quantity( 

166 1.0, mmunit.kilojoule_per_mole 

167 ), 

168 contrastGroup: t.Optional[t.Iterable[int]] = None, 

169 ) -> None: 

170 group1 = list(group1) 

171 group2 = list(group2) 

172 contrasting = contrastGroup is not None 

173 if contrasting: 

174 contrastGroup = list(contrastGroup) 

175 nonbondedForce = NonbondedForceSurrogate(nonbondedForce) 

176 cutoff = nonbondedForce.getCutoffDistance() 

177 expression = ( 

178 f"-(lj + coul){'*sign1*sign2' if contrasting else ''}/ref" 

179 "; lj = 4*epsilon*(1/y^2 - 1/y)" 

180 f"; coul = {ONE_4PI_EPS0}*q1q2*(1/x + (x^2 - 3)/2)" 

181 f"; x = r/{cutoff}" 

182 "; y = abs((r/sigma)^6 - 2) + 2" 

183 "; q1q2 = min(0, charge1*charge2)" 

184 "; epsilon = sqrt(epsilon1*epsilon2)" 

185 "; sigma = (sigma1 + sigma2)/2" 

186 "; ref = 1" 

187 ) 

188 super().__init__(expression) 

189 if nonbondedForce.usesPeriodicBoundaryConditions(): 

190 self.setNonbondedMethod(self.CutoffPeriodic) 

191 else: 

192 self.setNonbondedMethod(self.CutoffNonPeriodic) 

193 self.setCutoffDistance(cutoff) 

194 for parameter in ("charge", "sigma", "epsilon") + ("sign",) * contrasting: 

195 self.addPerParticleParameter(parameter) 

196 for atom in range(nonbondedForce.getNumParticles()): 

197 parameters = nonbondedForce.getParticleParameters(atom) 

198 if contrasting: 

199 parameters += (-1 if atom in contrastGroup else 1,) 

200 self.addParticle(parameters) 

201 for exception in range(nonbondedForce.getNumExceptions()): 

202 i, j, *_ = nonbondedForce.getExceptionParameters(exception) 

203 self.addExclusion(i, j) 

204 self.setUseSwitchingFunction(nonbondedForce.getUseSwitchingFunction()) 

205 self.setSwitchingDistance(nonbondedForce.getSwitchingDistance()) 

206 self.setUseLongRangeCorrection(False) 

207 self.addInteractionGroup(group1, group2) 

208 if contrasting: 

209 self.addInteractionGroup(group1, contrastGroup) 

210 if isinstance(reference, openmm.Context): 

211 reference = evaluate_in_context(self, reference) 

212 self.setEnergyFunction(expression.replace("ref = 1", f"ref = {reference}")) 

213 # pylint: disable=duplicate-code 

214 self._registerCV( 

215 mmunit.dimensionless, 

216 group1, 

217 group2, 

218 nonbondedForce, 

219 reference, 

220 contrastGroup, 

221 )
